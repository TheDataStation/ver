CREATE NODE TABLE Column(
id INT64, 
dbname STRING, 
path STRING, 
sourcename STRING, 
columnname STRING, 
datatype STRING, 
totalvalues INT64, 
uniquevalues INT64, 
nonemptyvalues INT64, 
entities STRING, 
minhash STRING, 
minvalue DOUBLE, 
maxvalue DOUBLE, 
avgvalue DOUBLE, 
median DOUBLE, 
iqr DOUBLE, 
PRIMARY KEY (id));

CREATE REL TABLE Edge(FROM Column TO Column, similar INT64)
